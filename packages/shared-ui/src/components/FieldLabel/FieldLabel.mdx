## Implementation

Usage example using jsx

```jsx
const Form: FC = () => {
  return
      <FieldLabel
        label="Username"
        htmlFor="username"
        description="You cannot change this later"
        tooltip="Your username is unique to you and will appear on your card"
      />
      <input
        className="h-8 px-2 rounded-md outline outline-1 border-colour-onSurface-outline dark:border-colour-onSurface-outline-dark"
        id="username"
        type="text"
        placeholder="Username"
      />
};
