## Implementation
Usage example using jsx

```jsx
const SomeForm: FC = () => {
  const [selected, setSelected] = useState(false);
  return <ToggleInput
           id="subscribeToggle"
           name="subscribe"
           selected={selected}
           onChange={() => setSelected(!selected)}
         />;
};
```
