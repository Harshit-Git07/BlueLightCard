## Implementation

```jsx
const Header: FC = () => {
  const props: NavBarProps = {
    links: [
      { id: '1', label: 'Link 1', url: '/' },
      { id: '2', label: 'Link 2', url: '/', links: [
        { id: '1', label: 'Sub Link 1, url: '/'' },
        { id: '2', label: 'Sub Link 2, url: '/'' }
      ] }
    ]
  };

  return <NavBar {...props} />;
};
```
